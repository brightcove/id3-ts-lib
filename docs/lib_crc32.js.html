/**
 * Compute a CRC32 value for a portion of a buffer and write the results back
 * into that buffer
 *
 * @param {Buffer} buffer - The buffer used as both source and destination
 * @param {Number} destination - The offset in the buffer where the resulting CRC32 is to be written
 * @param {Number} startOffset - The start of the sub-buffer to calulate the CRC32 value for (inclusive)
 * @param {Number} endOffset - The end of the sub-buffer to calculate the CRC32 for (exclusive)
 */
exports.crc32 = (buffer, destination, startOffset, endOffset) => {
  let crcVal = -1;
  let num = endOffset - startOffset;

  for (let i = 0; i < num; i++) {
    crcVal = CRC32_IEEE[(crcVal ^ buffer[i + startOffset]) & 0xff] ^ (crcVal >>> 8);
  }

  buffer.writeInt32LE(crcVal, destination);
};
